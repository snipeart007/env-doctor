### 2.2 Deterministic UID Generation System

**Task:** Implement consistent, collision-resistant UID generation

**Steps:**
1. Create `src/env_doctor/database/uid_generator.py`:
   ```python
   import hashlib
   from typing import Any
   import json
   
   def generate_uid(*components: Any) -> str:
       """
       Generate deterministic UID from components.
       
       Uses SHA-256 hash of sorted, JSON-serialized components.
       Ensures same inputs always produce same UID.
       """
       # Sort components for consistency
       sorted_components = sorted(str(c) for c in components)
       
       # Create stable string representation
       stable_repr = json.dumps(sorted_components, sort_keys=True)
       
       # Generate SHA-256 hash
       hash_obj = hashlib.sha256(stable_repr.encode('utf-8'))
       
       # Return hex digest
       return hash_obj.hexdigest()
   
   def generate_package_uid(package_name: str) -> str:
       """Generate UID for package."""
       return generate_uid("package", package_name.lower())
   
   def generate_version_uid(package_name: str, version: str) -> str:
       """Generate UID for package version."""
       return generate_uid("version", package_name.lower(), version)
   
   def generate_dependency_uid(
       package_name: str,
       package_version: str,
       dependency_name: str
   ) -> str:
       """Generate UID for dependency relationship."""
       return generate_uid(
           "dependency",
           package_name.lower(),
           package_version,
           dependency_name.lower()
       )
   
   def generate_compatibility_uid(
       package_name: str,
       package_version_range: str,
       dependency_name: str,
       dependency_version_range: str
   ) -> str:
       """Generate UID for compatibility rule."""
       return generate_uid(
           "compatibility",
           package_name.lower(),
           package_version_range,
           dependency_name.lower(),
           dependency_version_range
       )
   
   def generate_stack_uid(stack_name: str, cuda_version: str) -> str:
       """Generate UID for stable stack."""
       return generate_uid("stack", stack_name.lower(), cuda_version)
   
   def generate_wheel_uid(
       package_name: str,
       package_version: str,
       python_tag: str,
       platform_tag: str,
       cuda_variant: str = ""
   ) -> str:
       """Generate UID for wheel availability."""
       return generate_uid(
           "wheel",
           package_name.lower(),
           package_version,
           python_tag,
           platform_tag,
           cuda_variant
       )

       """Generate UID for runtime profile."""
   ```

   ```python
   from typing import Dict, List, Any
   from packaging.specifiers import SpecifierSet
   from packaging.version import Version
   import re
   
   class DataValidator:
       """Validates database entries for correctness."""
       
       @staticmethod
       def validate_package_name(name: str) -> bool:
           """Validate package name format."""
           # PEP 508 package name pattern
           pattern = r'^[a-zA-Z0-9]([a-zA-Z0-9._-]*[a-zA-Z0-9])?$'
           return bool(re.match(pattern, name))
       
       @staticmethod
       def validate_version(version: str) -> bool:
           """Validate version string."""
           try:
               Version(version)
               return True
           except Exception:
               return False
       
       @staticmethod
       def validate_version_range(version_range: str) -> bool:
           """Validate version specifier."""
           try:
               SpecifierSet(version_range)
               return True
           except Exception:
               return False
       
       @staticmethod
       def validate_compatibility_rule(rule: Dict[str, Any]) -> List[str]:
           """Validate compatibility rule. Returns list of errors."""
           errors = []
           
           # Required fields
           required = [
               "package_name",
               "package_version_range",
               "dependency_name",
               "dependency_version_range",
               "compatibility_type",
               "severity",
               "confidence_level"
           ]
           
           for field in required:
               if field not in rule:
                   errors.append(f"Missing required field: {field}")
           
           # Validate package names
           if not DataValidator.validate_package_name(rule.get("package_name", "")):
               errors.append("Invalid package_name")
           
           if not DataValidator.validate_package_name(rule.get("dependency_name", "")):
               errors.append("Invalid dependency_name")
           
           # Validate version ranges
           if not DataValidator.validate_version_range(rule.get("package_version_range", "")):
               errors.append("Invalid package_version_range")
           
           if not DataValidator.validate_version_range(rule.get("dependency_version_range", "")):
               errors.append("Invalid dependency_version_range")
           
           # Validate enums
           valid_types = ["compatible", "incompatible", "partial", "runtime-risk", "untested"]
           if rule.get("compatibility_type") not in valid_types:
               errors.append(f"Invalid compatibility_type. Must be one of: {valid_types}")
           
           valid_severities = ["critical", "high", "medium", "low", "info"]
           if rule.get("severity") not in valid_severities:
               errors.append(f"Invalid severity. Must be one of: {valid_severities}")
           
           valid_confidence = ["experimental", "community-tested", "stable", "production-tested"]
           if rule.get("confidence_level") not in valid_confidence:
   ```

### 7.2 Weight Memory Calculation

Calculate model weight memory:

**Formula:**
```
weight_memory = num_parameters × bytes_per_parameter
```

**Bytes per parameter:**
- FP32: 4 bytes
- FP16/BF16: 2 bytes
- INT8: 1 byte
- INT4: 0.5 bytes
- AWQ/GPTQ: ~0.5-1 bytes (depends on group size)

**Example:**
```python
# Llama-3-70B in FP16
params = 70_000_000_000
bytes_per_param = 2
weight_memory = params * bytes_per_param / (1024**3)  # GB
# Result: ~130 GB
```

### 7.3 KV Cache Estimation

Estimate KV cache memory:

**Formula:**
```
kv_cache = batch_size × seq_length × hidden_size × num_layers × 2 × dtype_bytes
```

**Factors:**
- `batch_size` - Number of concurrent sequences
- `seq_length` - Maximum sequence length
- `hidden_size` - Model hidden dimension
- `num_layers` - Number of transformer layers
- `2` - Key and Value caches
- `dtype_bytes` - Data type size

**Example:**
```python
# Llama-3-70B, batch=1, seq=4096, FP16
batch = 1
seq_len = 4096
hidden = 8192
layers = 80
dtype_bytes = 2

kv_cache = batch * seq_len * hidden * layers * 2 * dtype_bytes / (1024**3)
# Result: ~10 GB
```

### 7.4 Runtime Overhead Profiles

Model runtime-specific overhead:

**Runtime Profiles:**
```python
{
  "transformers": {
    "kv_overhead_multiplier": 1.0,
    "fragmentation_multiplier": 1.2,
    "activation_multiplier": 1.5
  },
  "vllm": {
    "kv_overhead_multiplier": 1.1,
    "fragmentation_multiplier": 1.15,
    "activation_multiplier": 1.2
  },
  "llama.cpp": {
    "kv_overhead_multiplier": 0.9,
    "fragmentation_multiplier": 1.05,
    "activation_multiplier": 1.0
  }
}
```

**Apply:**
```python
total_vram = (
    weight_memory +
    kv_cache * kv_overhead_multiplier +
    activation_memory * activation_multiplier
) * fragmentation_multiplier
```

### 7.5 OOM Risk Detection

**Risk Levels:**
- **Safe** - <70% VRAM usage
- **Warning** - 70-85% VRAM usage
- **Danger** - 85-95% VRAM usage
- **Critical** - >95% VRAM usage

**Factors:**
- Available VRAM (from GPU detection)
- Estimated usage
- Safety margin (10-20%)
- Runtime overhead
- Fragmentation

**Warnings:**
- Suggest smaller batch size
- Suggest shorter sequence length
- Suggest quantization
- Suggest gradient checkpointing
- Suggest model sharding

```
kv_cache = batch × seq_len × hidden × layers × 2 × dtype_bytes
```
